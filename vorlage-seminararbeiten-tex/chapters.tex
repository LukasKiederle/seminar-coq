\begin{lstlisting}[language=coq,firstnumber=1,caption=Coq Beispiel,label=lst:api-config]
	Inductive day : Type :=
	| monday
	| tuesday
	| wednesday
	| thursday
	| friday
	| saturday
	| sunday.

	Definition next_weekday (d:day) : day :=
	match d with
	| monday => tuesday
	| tuesday => wednesday
	| wednesday => thursday
	| thursday => friday
	| friday => monday
	| saturday => monday
	| sunday => monday
	end.

	Compute (next_weekday friday).
	(* ==> monday : day *)
	Compute (next_weekday (next_weekday saturday)).
	(* ==> tuesday : day *)
\end{lstlisting}
